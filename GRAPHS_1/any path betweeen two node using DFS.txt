// finding any path between two vertex using dfs 

unordered_set<int> s; // for storing the visited node 

bool dfs(int curr, int end){
  if(curr==end) return true;
  s.insert(curr);  // mark the visited 
  for(auto neighbour: graph[curr]){   // find all neighbour 
    if(not s.count(neighbour)){   // is neighbour is not visited 
      int result = dfs(neighbour,end);  
      if(result) return true;
    }
  }
  return false;
}
bool anypath(int src, int dst){
  return dfs(src, dst);
}
