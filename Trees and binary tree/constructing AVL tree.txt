// node class has height variable also to maintain its height
class node{
  public:
  int val;
  int height;
  node* right;
  node* left;
  node(int n){
    this->val= n;
    height=1;
    left=right=NULL;
  }
};

int getheight(node* root){
  if(root==NULL) return 0;
  else return root->height;
}


// right rotation (L L)

node* rightrotation(node* root){
  node* child= root->left;
  node* childright= child->right;
  child->right= root;
  root->left= childright;
  root->height= 1+ max(getheight(root->left), getheight(root->right));
  child->height= 1+ max(getheight(child->left), getheight(child->right));
  
  return child;
}

// left rotation (R R)

node* leftrotation(node* root){
  node* child= root->right;
  node* childleft= child->left;
  child->left= root;
  root->right= childleft;
  root->height= 1+ max(getheight(root->left), getheight(root->right));
  child->height= 1+ max(getheight(child->left), getheight(child->right));
  
  return child;
}

// making avl tree (insertion )
node* insert(node* root, int key){
  
  // filling of node (neeche gaye )
  if(root==NULL)
    return new node(key);
    
    if(root->val< key){
      root->right= insert(root->right, key);
    }
    else if(root->val>key){
      root->left= insert(root->left, key);
    }
    else return root;  // duplicate value not allowed 
    
    // ab uper aayenge 
    
    //1-> pdate height

  root->height= 1+ max(getheight(root->left), getheight(root->right)); 
  
    // 2-> balancing check 
    
    int balance = getheight(root->left)- getheight(root->right);
    
    // left left 
    if(balance>1 && root->left->val>key){
       return rightrotation(root);
    }
    // left right
    else if(balance>1 && root->left->val<key){
      root->left= leftrotation(root->left);
      return rightrotation(root);
    }
    // right right 
    else if(balance<-1 && root->right->val>key){
      return leftrotation(root);
    }
    // right left 
    else if(balance<-1 && root->right->val>key){
     root->right =  rightrotation(root->right);
      return leftrotation(root);
      
    }
    else return root;
    
    
    
}
