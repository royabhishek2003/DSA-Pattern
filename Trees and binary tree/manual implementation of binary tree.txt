// ceating a tree node

class node{
  public:
  int val;
  node* left;
  node* right;
  node(int val){
    this->val=val;
    left=right=NULL;
  }
};

// to display a binary trees

void display(node* root){
  if(root==NULL) return;
  cout<<root->val<<" ";
  display(root->left);
  display(root->right);
}

// finding the sum of a tree node

int sumtree(node* root){
  if(root==NULL) return 0;
  return root->val + sumtree(root->left) + sumtree(root->right);
}

// finding the size of binary trees

int size(node* root){
  if(root==NULL) return 0;
  return 1+ size(root->left) + size(root->right);
}

// finding maxvalue node of a binary trees

int maxvalue(node* root){
  if(root==NULL) return INT_MIN;
  return max(root->val,max(maxvalue(root->left), maxvalue(root->right)));
}

// finding the product of all the nodes

int product(node* root){
  if(root==NULL) return 1;
  return root->val * product(root->left) * product(root->right);
}

// finding the level of the binary trees!

int level(node* root){
  if(root==NULL) return 0;
 return  1+ max(level(root->left),level(root->right));
}
