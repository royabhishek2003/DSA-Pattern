// to traversal in a binary trees


// preorder traversal 

void preorder(node* root){
  if(root==NULL) return;
  cout<<root->val<<" ";
  preorder(root->left);
  preorder(root->right);
}

// inorder traversal 

void inorder(node* root){
  if(root==NULL) return;
  inorder(root->left);
  cout<<root->val<<" ";
  inorder(root->right);
}

// postorder traversal 

void postorder(node* root){
  if(root==NULL) return;
  postorder(root->left);
  postorder(root->right);
  cout<<root->val<<" ";
}


 //  nth level order  traversl (left->right)

void printnthlevel(node* root, int current, int level){
  if(root==NULL ) return ;
  if(current==level) {
    cout<<root->val<<" ";
    return;
  }
  printnthlevel(root->left, current+1, level);
  printnthlevel(root->right, current+1, level);
}


//  Nth level order traversal form right to left 
  
  void nthlevelrev(node* root, int current, int level){
    if(root==NULL) return;
    if(current==level) {
      cout<<root->val<<" ";
      return;
    }
    nthlevelrev(root->right, current+1, level);
    nthlevelrev(root->left, current+1,level);
   
  }

void levelorderqueue(node* root){
  queue<node*> q;
  q.push(root);
  while(q.size()>0){
    node* temp=q.front();
    q.pop();
    cout<<temp->val<<" ";
    if(temp->left!=NULL) q.push(temp->left);
    if(temp->right!=NULL) q.push(temp->right);
  }
