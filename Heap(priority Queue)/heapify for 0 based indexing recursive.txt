void heapify(vector<int>&v, int n, int i){
  int largest= i;
  int left= 2*i+1;
  int right= 2*i+2;
  if(left<n && v[largest]<v[left]){
    largest=left;
  }
  if(right<n && v[largest]< v[right]){
    largest =right;
  }
  if(i!=largest){
    swap(v[i], v[largest]);
    heapify(v, n, largest);
  }
}

int main() 
{
    vector<int> v= {2,3,1,2,6,3,7};
    int n= v.size();
    for(int i=n/2-1;i>=0;i--){
      heapify(v, n, i);
    }
    
    for(int ele: v) cout<<ele<<" ";
    return 0;
}
