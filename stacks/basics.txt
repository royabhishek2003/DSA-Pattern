 // display the stack recursively in reverse order

void displayrecrev(stack<int> &st){
    if(st.size()==0) return;
    cout<<st.top()<<" ";
    int x=st.top();
    st.pop();
    displayrecrev(st);
    st.push(x);
}

// display the stack by recursion 

void displayrec(stack<int> &st){
    if(st.size()==0) return;
    int x=st.top();
    st.pop();
    displayrec(st);
    st.push(x);
    cout<<x<<" ";
}

// push an element at the bootom of a stack using recursion

void pushatbottomrec(stack<int> &st, int num){
    if(st.size()==0){
        st.push(num);
        return;
    }
    int x=st.top();
    st.pop();
    pushatbottomrec(st, num);
    st.push(x);
}
